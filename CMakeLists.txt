cmake_minimum_required(VERSION 3.10)
project(MarkdownEditor CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include directories
include_directories(include)

# Add source files
set(SOURCE_FILES
    src/main.cpp
    src/editor.cpp
    src/parser.cpp
    src/renderer.cpp
)

# Create the executable
add_executable(markdown-editor ${SOURCE_FILES})
